          Upload JUnit XML Reports

          Upload JUnit XML test reports from your CI pipeline (Jenkins, GitHub Actions, GitLab CI, etc.) to detect flaky tests across multiple runs.

          📁

          Drag & drop JUnit XML files here, or browse

          Accepts multiple .xml files from different test runs

          CI Pipeline Setup

          GitHub Actions

          # .github/workflows/ci.yml
- name: Run tests with JUnit output
  run: pytest --junitxml=test-results/junit.xml

# Upload to FlakyScan
- name: Report to FlakyScan
  run: |
    curl -X POST http://<flakyscan-host>/ingest-junit \
      -F "files=@test-results/junit.xml"

          Jenkins

          // Jenkinsfile
stage('Report Flaky Tests') {
    steps {
        sh '''
            curl -X POST http://<flakyscan-host>/ingest-junit \
              -F "files=@test-results/junit.xml"
        '''
    }
}

          GitLab CI

          # .gitlab-ci.yml
report_flaky:
  script:
    - pytest --junitxml=junit.xml || true
    - curl -X POST http://$FLAKYSCAN_HOST/ingest-junit
        -F "files=@junit.xml"
  artifacts:
    reports:
      junit: junit.xml
